#include <stdio.h>
 
int main() {
    int a,b;
    
    printf("Ingrese el primer numero: ");
    scanf("%d", &a);
    
    printf("Ingrese el segundo numero: ");
    scanf("%d", &b);
    
    if (a > b){
        printf("El numero mayor es: %d. \n", a);
    }
    else if(b > a){
        printf("El numero es mayor es: %d. \n", b);
    }
    else{
        printf("Ambos numeros son iguales.\n");
    }
    return 0;
}
