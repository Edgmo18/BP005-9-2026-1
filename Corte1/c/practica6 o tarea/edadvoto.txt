#include<stdio.h>
#include<stdbool.h>

int main(){
    int edad;
    bool esciudadano = true;
    
    printf("Ingrese su edad\n");
    scanf("%d" , &edad);
    
    if (edad >= 18){
        printf("Está habilitado para votar\n");
    
       if (esciudadano){
          printf("Usted es ciudadano, así que puede votar!\n");
       }
       else {
          printf("Debe ser un ciudadano para votar.\n");
        }
    }
    
    else {
       printf("No tiene la edad suficiente para votar.\n");
    }
}
