#include <iostream>
using namespace std;

int main(){
    int opcion;

    do{
      cout << endl <<"MENU"<<endl;
      cout <<"1. Saludar"<<endl;
      cout <<"2. Despedirse"<<endl;
      cout <<"0. Salir"<<endl;
      cout <<"Seleccione una opcion: "<<endl;
      cin >> opcion;

      if (opcion == 1){
         cout <<"Hola."<< endl;
      }
      else if (opcion == 2){
         cout <<"Hasta luego."<<endl;
      }
      else if (opcion == 0){
         cout <<"Fin del programa."<<endl;
      }
      else{
         cout <<"Opcion invalida."<<endl;
      } 
    
    }

    while (opcion != 0);
  
   return 0;
}
