#include <stdio.h>

void normal(int a, int b){
    printf("Orden normal: A = %d, B = %d\n", a, b);
    }

void intercambiar(int a, int b) {
    printf("Orden inverso: A = %d, B = %d\n", b, a);
    }

int main(){
    int x = 10;
    int y = 20;
    normal(x, y);
    intercambiar(x, y);
    return 0;
    }
